# patches/vikit_common_CMakeLists.txt
# Drop-in replacement for rpg_vikit/vikit_common/CMakeLists.txt
#
# Fixes applied vs original:
#   1. cmake_minimum_required moved before project()
#   2. x86 SIMD flags (-mmmx -msse -msse2 -msse3 -mssse3) removed
#   3. catkin dependency removed — pure ament_cmake
#   4. Sophus added via target property (ros-humble-sophus uses modern CMake)
#   5. ament_export_targets added for correct downstream linking

cmake_minimum_required(VERSION 3.10)
project(vikit_common)

set(CMAKE_BUILD_TYPE Release)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_POSITION_INDEPENDENT_CODE ON)

if(CMAKE_SYSTEM_PROCESSOR MATCHES "aarch64")
  set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -O3 -mcpu=native -ffast-math")
else()
  set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -O3 -march=native")
endif()

find_package(ament_cmake REQUIRED)
find_package(Eigen3 REQUIRED)
find_package(OpenCV REQUIRED)
find_package(Sophus REQUIRED)

file(GLOB VIKIT_SRCS src/*.cpp)

add_library(vikit_common SHARED ${VIKIT_SRCS})

target_include_directories(vikit_common PUBLIC
  $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
  $<INSTALL_INTERFACE:include>
  ${EIGEN3_INCLUDE_DIR}
  ${OpenCV_INCLUDE_DIRS}
  $<TARGET_PROPERTY:Sophus::Sophus,INTERFACE_INCLUDE_DIRECTORIES>
)

target_link_libraries(vikit_common
  ${OpenCV_LIBS}
  Eigen3::Eigen
  Sophus::Sophus
)

install(TARGETS vikit_common
  EXPORT vikit_common_export
  ARCHIVE DESTINATION lib
  LIBRARY DESTINATION lib
  RUNTIME DESTINATION bin
)

install(DIRECTORY include/
  DESTINATION include
)

ament_export_targets(vikit_common_export HAS_LIBRARY_TARGET)
ament_export_include_directories(include)
ament_export_dependencies(Eigen3 OpenCV Sophus)

ament_package()
